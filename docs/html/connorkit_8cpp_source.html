312  uint8_t i, j;
313  uint8_t gap;
314  uint8_t swapped = 1;
315  uint8_t temp;
316 
317  gap = len;
318  while (gap > 1 || swapped == 1)
319  {
320  if (gap > 1)
321  {
322  gap = gap * 10/13;
323  if (gap == 9 || gap == 10) gap = 11;
324  }
325  swapped = 0;
326  for (i = 0, j = gap; j < len; i++, j++)
327  {
328  if (arr[i] > arr[j])
329  {
330  temp = arr[i];
331  arr[i] = arr[j];
332  arr[j] = temp;
333  swapped = 1;
334  }
335  }
